Cada vez que recibas una foto tendra la informacion de las farmacias de turno de una determinada ciudad. Deberas transformar esa informacion a un JSON con el siguiente formato:
{
    {mes}: {
        {ciudad}: {
            "fuente": {link de la fuente },
            "confianza": {escala de confiabilidad del 1 al 3},
            "dias": {
                "1": [
                    {
                        "nombre": {nombre farmacia},
                        "direccion": {direccion farmacia},
                        "telefono": {telefono farmacia},
                        "mapa": {url en google maps de la farmacia}
                    }
                ],
                "2": [
                    {
                        "nombre": {nombre farmacia},
                        "direccion": {direccion farmacia},
                        "telefono": {telefono farmacia},
                        "mapa": {url en google maps de la farmacia}
                    }
                ],

{
    "agosto": {
        "Glew": {

            "confianza": 3,
            "dias": {
